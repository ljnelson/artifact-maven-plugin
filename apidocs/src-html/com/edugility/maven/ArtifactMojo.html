061/**
062 * A mojo that sorts Maven artifacts topologically and then arranges
063 * the processing of them with a user-supplied processor.
064 *
065 * @author <a href="http://about.me/lairdnelson"
066 * target="_parent">Laird Nelson</a>
067 *
068 * @see ArtifactsProcessor
069 */
070@Mojo(name = "process")
071public class ArtifactMojo extends AbstractMojo {
072
073  /**
074   * The {@link MavenProject} in effect.
075   */
076  @Parameter(defaultValue = "${project}", readonly = true, required = true)
077  private MavenProject project;
078
079  /**
080   * The {@link DependencyGraphBuilder} used to collect all
081   * dependencies of a project.
082   */
083  @Component
084  private DependencyGraphBuilder dependencyGraphBuilder;
085
086  /**
087   * The {@link ArtifactResolver} to use when a given artifact has not
088   * yet been resolved.
089   */
090  @Component
091  private ArtifactResolver resolver;
092
093  /**
094   * An {@link ArtifactsProcessor} implementation that will receive a
095   * topologically sorted collection of resolved Maven artifacts.
096   */
097  @Parameter(required = true)
098  private ArtifactsProcessor artifactsProcessor;
099
100  /**
101   * The local {@link ArtifactRepository} in effect.
102   */
103  @Parameter(defaultValue = "${localRepository}", readonly = true, required = true)
104  private ArtifactRepository localRepository;
105
106  /**
107   * An {@link ArtifactFilter} for choosing what {@link Artifact}s are
108   * to be processed.
109   */
110  @Parameter
111  private ArtifactFilter artifactFilter;
112
113  /**
114   * Creates a new {@link ArtifactMojo}.
115   */
116  public ArtifactMojo() {
117    super();
118  }
119
120  /**
121   * Collects all {@link Artifact}s that should be
122   * processed&mdash;according to the {@linkplain #artifactFilter
123   * associated <code>ArtifactFilter</code>}&mdash;and hands them off
124   * to the associated {@link #artifactsProcessor ArtifactsProcessor}.
125   *
126   * @exception MojoExecutionException if processing failed
127   *
128   * @exception MojoFailureException if setup failed
129   *
130   * @see ArtifactsProcessor
131   */
132  @Override
133  public void execute() throws MojoExecutionException, MojoFailureException {
134    if (this.project == null) {
135      throw new MojoFailureException("this.project == null", new IllegalStateException("this.project == null", new NullPointerException("this.project")));
136    }
137    if (this.dependencyGraphBuilder == null) {
138      throw new MojoFailureException("this.dependencyGraphBuilder == null", new IllegalStateException("this.dependencyGraphBuilder == null", new NullPointerException("this.dependencyGraphBuilder")));
139    }
140
141    final ArtifactsProcessor processor = this.getArtifactsProcessor();
142    if (processor == null) {
143      throw new MojoFailureException("this.getArtifactsProcessor() == null", new IllegalStateException("this.getArtifactsProcessor() == null", new NullPointerException("this.getArtifactsProcessor()")));
144    }
145
146    Collection<? extends Artifact> artifacts = null;
147
148    try {
149      artifacts = new Artifacts().getArtifactsInTopologicalOrder(this.project, 
150                                                                 this.dependencyGraphBuilder,
151                                                                 this.artifactFilter,
152                                                                 resolver,
153                                                                 localRepository);
154    } catch (final DependencyGraphBuilderException wrapMe) {
155      throw new MojoExecutionException("Failed to get dependency artifacts", wrapMe);
156    } catch (final ArtifactResolutionException wrapMe) {
157      throw new MojoExecutionException("Failed to get dependency artifacts", wrapMe);
158    }
159
160    if (artifacts != null && !artifacts.isEmpty()) {
161      try {
162        processor.process(this.project, artifacts, this.getLog());
163      } catch (final ArtifactsProcessingException wrapMe) {
164        throw new MojoExecutionException("Failed to process artifacts", wrapMe);
165      }
166    }
167  }
168
169  /**
170   * Returns an {@link ArtifactsProcessor} for use by the {@link
171   * #execute()} method.
172   *
173   * <p>This method may return {@code null}.</p>
174   *
175   * @return an {@link ArtifactsProcessor}
176   *
177   * @see ArtifactsProcessor
178   */
179  public ArtifactsProcessor getArtifactsProcessor() {
180    return this.artifactsProcessor;
181  }
182
183}
